## 核心组件

### 1. Router（路由）

```python
class MoERouter:
    def __init__(self, num_experts, top_k=2):
        self.num_experts = num_experts
        self.top_k = top_k
        self.routing_network = Linear(num_experts)
    
    def forward(self, x):
        # 计算每个专家的权重
        logits = self.routing_network(x)
        # 选择 top-k 专家
        top_k_logits, top_k_indices = torch.topk(logits, self.top_k)
        # 归一化
        weights = F.softmax(top_k_logits, dim=-1)
        return weights, top_k_indices
```

### 2. Expert（专家）

```python
class Expert(nn.Module):
    def __init__(self, input_dim, output_dim):
        super().__init__()
        self.network = nn.Sequential(
            nn.Linear(input_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, output_dim)
        )
    
    def forward(self, x):
        return self.network(x)
```

### 3. Aggregator（聚合）

```python
class MoEAggregator:
    def aggregate(self, outputs, weights):
        # 加权聚合
        weighted_outputs = [w * o for w, o in zip(weights, outputs)]
        return sum(weighted_outputs)
```
