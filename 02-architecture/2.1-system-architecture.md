## 核心模块

### 任务管理器

```python
class TaskManager:
    """任务管理器"""
    
    def __init__(self):
        self.task_queue = PriorityQueue()
        self.task_registry = {}
        self.agent_pool = AgentPool()
    
    def create_task(self, task_spec):
        task = Task(
            id=self.generate_id(),
            spec=task_spec,
            status=TaskStatus.PENDING
        )
        self.task_registry[task.id] = task
        return task
    
    def process_task(self, task_id):
        task = self.task_registry[task_id]
        subtasks = self.decompose_task(task)
        results = [agent.execute(subtask) for subtask in subtasks]
        return self.integrate_results(results)
```

### Agent 池管理

```python
class AgentPool:
    """Agent 池：管理可用的 Agent 实例"""
    
    def __init__(self):
        self.agents = {}
        self.capabilities = {}
    
    def select_agent(self, requirements):
        capable = self.capabilities.get(requirements.capability, [])
        available = [a for a in capable if a.is_available()]
        return self.load_balancer.select(available)
```
